# Regression Models
def train_regression_model(model, X_train, X_test, y_train, y_test):
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)
    mse = mean_squared_error(y_test, y_pred)
    r2 = r2_score(y_test, y_pred)
    return mse, r2

# Linear Regression

# Classification Models

# Convert quality scores to categories
def quality_to_category(quality):
    if quality <= 5:
        return 'low'
    elif quality <= 7:
        return 'medium'
    else:
        return 'high'

y_cat = wines['quality'].apply(quality_to_category)
y_train_cat = y_train.apply(quality_to_category)
y_test_cat = y_test.apply(quality_to_category)

def train_classification_model(model, X_train, X_test, y_train, y_test):
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)
    accuracy = accuracy_score(y_test, y_pred)
    return accuracy, classification_report(y_test, y_pred), confusion_matrix(y_test, y_pred)

# Logistic Regression

from sklearn.neighbors import NearestNeighbors

# Select features for recommendation
features = ['fixed acidity', 'volatile acidity', 'citric acid', 'residual sugar', 
            'chlorides', 'free sulfur dioxide', 'total sulfur dioxide', 'density', 
            'pH', 'sulphates', 'alcohol']

# Prepare the feature matrix
X = wines[features]

# Normalize the features
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Create and fit the k-NN model
k = 5  # Number of neighbors to consider
model = NearestNeighbors(n_neighbors=k, metric='euclidean')
model.fit(X_scaled)

def get_wine_recommendations(query_wine_index, num_recommendations=5):
    # Get the features of the query wine
    query_features = X_scaled[query_wine_index].reshape(1, -1)
    
    # Find the k nearest neighbors
    distances, indices = model.kneighbors(query_features, n_neighbors=num_recommendations+1)
    
    # Exclude the query wine itself
    recommended_indices = indices.flatten()[1:]
    
    # Get the recommended wines
    recommended_wines = wines.iloc[recommended_indices]
    
    return recommended_wines[['type', 'quality'] + features]

# Example usage
query_wine_index = 100  # Index of the wine for which we want recommendations
recommendations = get_wine_recommendations(query_wine_index)
print(recommendations)
